let USD_ILS_RATE = 3.02;

const COMMISSION = 0.02;

async function fetchUsdIlsRate() {

try {

const res = await fetch('https://api.exchangerate.host/latest?base=USD&symbols=ILS');

const data = await res.json();

if (data && data.rates && data.rates.ILS) {

USD_ILS_RATE = data.rates.ILS;

}

} catch (e) {

console.log('Using fallback rate');

}

updateRateDisplays();

}

function getSellRate() {

return USD_ILS_RATE * (1 - COMMISSION);

}

function getBuyRate() {

return USD_ILS_RATE * (1 + COMMISSION);

}

document.getElementById('rateDisplay').textContent = getSellRate().toFixed(2);

document.getElementById('rateDisplay2').textContent = getBuyRate().toFixed(2);

}

function processSell() {

const amount = parseFloat(document.getElementById('sellAmount').value);

const wallet = document.getElementById('sellWallet').value;

if (!amount || amount < 1) {

alert('الرجاء إدخال مبلغ صالح (على الأقل 1 USDT)');

return;

}

if (!wallet) {

alert('الرجاء إدخال معلومات الاستقبال');

return;

}

document.getElementById('depositAddress').textContent = MY_USDT_WALLET;

document.getElementById('sellSection').classList.remove('active');

document.getElementById('sellResult').classList.add('active');

document.getElementById('displaySellAmount').textContent = amount;

document.getElementById('displayReceiveAmount').textContent = (amount * getSellRate()).toFixed(2);

document.getElementById('displaySellWallet').textContent = wallet;

}
